326 double DecisionTreeClassifier::impurityEval1(const std::vector<attr_set*>* dec_set)
327 {
328  double entropy = 0.0;
329  double max = 0.0;
330  double total = 0.0;
331  double prob = 0.0;
332  class_freq_map.clear();

334  {

338  {
339  string str_attr(attr_it->second);
340  int freq =class_freq_map[str_attr];

345

347  {
348  total += it2->second;
349  }
350  for(std::tr1::unordered_map<string, int>::iterator it3=class_freq_map.begin(); it3 != class_freq_map.end(); it3++)
