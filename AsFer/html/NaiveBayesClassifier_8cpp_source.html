201 void NaiveBayesClassifier::doNBMultinomialClassification(std::list<int>& corpus) //test set is corpus
202 {
203 
204  std::set<comparefunctor> topics; //classes
205  ifstream topicstxt;
206  topicstxt.open("topics.txt", ifstream::in);
207  char line[256];
208  topicstxt.getline(line, 256);
209  char *token;
210  double maxprobability = 0.0;
211  double Pr_dgivenc = 0.0;
212  double Pr_cgivend = 0.0;
213  double Pr_tigivenc = 0.0;
214  double Pr_c = 0.0;
215  char* maxtopic = 0;
216  double productofpr_tigivenc = 1.0;
217  while(topicstxt.good())
218  {
219  token = strtok(line, "~");
220  topics.insert(strdup(token));
221  topicstxt.getline(line, 256);
222  }
223 
224  std::list<int>* trset = readTrainingSet();
225  for(std::list<int>::iterator it = corpus.begin(); it != corpus.end(); it++)
226  {
227  //find probability of article being present in all classes - 120 classes
228  for(std::set<comparefunctor>::iterator topic_it = topics.begin(); topic_it != topics.end(); topic_it++)
229  {
230  //printf("topic = %s \n", topic_it->str);
231  std::list<int>* ar_c = getArticlesOfSingleClass(topic_it->str); //a priori probability
232  Pr_c = (double) ar_c->size() / trset->size(); //total no of articles in training set is 9603
233  std::list<tok_occur*>* ar_tokens = getTokensInArticle(*it);
234  for(std::list<tok_occur*>::iterator token_it = ar_tokens->begin(); token_it != ar_tokens->end(); token_it++)
235  {
236  //printf("token = %s \n", (*token_it)->word);
237  double occurrences = getOccurrences((*token_it)->word, ar_c);
238  double totaloccurrences = getTotalOccurrences(ar_c);
239  double vocabularysize = getVocabularySize();
240  double numerator = (double) (occurrences + 1.0); //laplace smoothing
241  double denominator = (double) (totaloccurrences + vocabularysize);
242  for(int i=0; i < (*token_it)->occur; i++)
243  productofpr_tigivenc = (double) (productofpr_tigivenc * numerator) / denominator;
244  }
245  Pr_dgivenc = (double) Pr_c * productofpr_tigivenc;
246  if(Pr_dgivenc > maxprobability)
247  {
248  maxprobability = Pr_dgivenc;
249  maxtopic = strdup(topic_it->str);
250  }
251 
252  //delete lists
253  for(std::list<tok_occur*>::iterator del_it = ar_tokens->begin(); del_it != ar_tokens->end(); del_it++)
254  {
255  delete ((*del_it)->word);
256  }
257  delete ar_tokens;
258  delete ar_c;
259  Pr_dgivenc = 0.0;
260  Pr_cgivend = 0.0;
261  Pr_tigivenc = 0.0;
262  Pr_c = 0.0;
263  productofpr_tigivenc = 1.0;
264  }
265  printf("article id %d is NaiveBayesMultiNomial-Classified in topic %s \n", *it, maxtopic);
266  maxtopic = 0;
267  maxprobability = 0.0;
268  }
269  topicstxt.close();
270 }
