51 std::list<weighted_asferrule>* SVMRetriever::get_rules(std::string query)
52 {
53  create_class_models();
54  write_test_dat();
55  classify_corpus();
56  char* query_class = classify_query(query);
57  std::list<asferrule>* rules = find_relevant_rules_for_query_class(query_class);
58 
59  std::list<weighted_asferrule> weighted_rules;
60  for(std::list<asferrule>::iterator it= rules->begin(); it != rules->end(); it++)
61  {
62  weighted_asferrule w;
63  w.asr = &(*it);
64  weighted_rules.push_back(w);
65  }
66 
67  return &weighted_rules;
68 }

70 std::list<asferrule>* SVMRetriever::find_relevant_rules_for_query_class(char* query_class)
71 {
72  int threshold = 0;
73  std::list<asferrule> rules;
74  ifstream svm_output;
75  char line[256];
76  char* tail;
77  char* svm_output_file = strcat("output_", query_class);
78  svm_output.open(svm_output_file);
79  std::list<asferrule>& corpus = index->get_rule_corpus();
80  std::list<asferrule>::iterator corpus_it = corpus.begin();
81  while(svm_output.good() && corpus_it != corpus.end())
82  {
83  svm_output.getline(line, 256);
84  int relevance = strtol(line, &tail, 0);
85  if(relevance > threshold)
86  rules.push_back(*corpus_it);
87  corpus_it++;
88  }
89  return &rules;
90 }

92 char* SVMRetriever::find_query_class()
93 {
94  int maximum = -10000;
95  char line[256];
96  char* tail;
97  char* query_class;
98  char* classes[] = {"tanu", "dhana", "sahaja", "matru", "putra", "roga", "yuti", "rina", "dharma", "karma", "labha", "vyaya"};
99  for(int i=0; i < 12; i++)
100  {
101  ifstream svm_output;
102  char* svm_output_file = strcat("query_output_", classes[i]);
103  svm_output.open(svm_output_file);
104  svm_output.getline(line, 256);
105  int relevance = strtol(line, &tail, 0);
106  if(relevance > maximum)
107  {
108  maximum = relevance;
109  query_class = classes[i];
110  }
111  }
112  return query_class;
113 
114 }
