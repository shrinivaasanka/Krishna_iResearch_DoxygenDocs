36 /*
37 Expirable template class implementation
38 ---------------------------------------
39 Similar to weak_ptr and shared_ptr, this is a generic datatype that wraps any datatype
40 and sets an expiry count to it. For example, a bitmap or a JPEG image can be wrapped in
41 expirable container and it can be access-controlled. If expiry count is 1, the object (an
42 image for example) can be written to or displayed only once and thus a singleton. Presently
43 this is only for rvalues. Implementing for lvalues (just a usage without assignment should
44 be able expire an object) seems to be a hurdle as there is no operator overloading available
45 for lvalues.
46 */

49 template <class T>
50 class expirable
51 {
52  T* data;
53  int expiry_count=6;
54  int assign_refcount=1;
55  int access_refcount=1;
56  int copy_assignment=1;
57  int move_assignment=1;
58 public:
59  expirable(T* x)
60  {
61  data=x;
62  }
63 
64  /*
65  Overloaded dereferencing operator
66  */
67  T& operator* () // const
68  {
69  cout<<"access_refcount: dereference operator invoked"<<endl;
70  if(access_refcount < expiry_count)
71  {
72  access_refcount++;
73  cout<<"access_refcount < expiry_count : dereference operator invoked"<<endl;
74  }
75  else
76  {
77  cout<<"access_refcount > expiry_count : this object expires, cannot be accessed anylonger"<<endl;
78  }
79  return *data;
80  }
81 
82  /*
83  Overloaded Copy Assignment operator
84  */
85 
86  T& operator=(expirable<T>& lvalue)
87  {
88  if(copy_assignment)
89  {
90  if(assign_refcount < expiry_count)
91  {
92  cout<<"copy_assignment: assignment operator invoked"<<endl;
93  assign_refcount++;
94  std::swap(data,lvalue.data);
95  }
96  else
97  {
98  cout<<"copy_assignment: this object expires, cannot be assigned to rvalue anylonger"<<endl;
99  //delete &lvalue;
100  }
101  }
102  return *data;
103  }
104 
105  /*
106  Overloaded Move Assignment operator
107  */
108 
109  T& operator=(expirable<T>&& rvalue)
110  {
111  if(move_assignment)
112  {
113  if(assign_refcount < expiry_count)
114  {
115  cout<<"move_assignment: assignment operator invoked"<<endl;
116  assign_refcount++;
117  data=std::move(rvalue.data);
118  }
119  else
120  {
121  cout<<"move_assignment: this object expires, cannot be assigned to rvalue anylonger"<<endl;
122  //delete &rvalue;
123  }
124  }
125  return *data;
126  }
127 };
