68 void asferindexer::create_index()
69 {
70  for(std::list<asferrule>::iterator corpus_iter = rule_corpus.begin(); corpus_iter != rule_corpus.end(); corpus_iter++)
71  {
72  asferrule rule = *corpus_iter;
73  rule.make_VSR(); //VSR - vector space representation in std::map
74  asferruleVSR& rule_vector = rule.asferruleVSR_map;
75  //std::list<string> rule_vector_tokens = get_tokens(rule_vector);
76 
77  for(asferruleVSR::iterator rule_vsr_iter = rule_vector.begin(); rule_vsr_iter != rule_vector.end(); rule_vsr_iter++)
78  {
79  weighted_term wt;
80  wt.token = new std::string(rule_vsr_iter->first);
81  wt.idf = 0.0;
82  //inv_index_type::iterator index_iter = inverted_index.find(wt);
83  inv_index_type::iterator index_iter;
84  for(index_iter = inverted_index.begin(); index_iter != inverted_index.end(); index_iter++)
85  {
86  if((*index_iter).first.token->compare(*(wt.token)) == 0)
87  break;
88  }
89 
90  if(index_iter == inverted_index.end())
91  {
92  std::list<weighted_asferrule> rulelist;
93  weighted_asferrule wa;
94  wa.asr = &(*corpus_iter);
95  wa.tf = rule_vsr_iter->second.no_of_occurrences;
96  rulelist.push_back(wa);
97  //inverted_index.insert(term_rulelist_pair(wt, rulelist));
98  inverted_index[wt] = rulelist;
99  }
100  else
101  {
102  weighted_asferrule wa;
103  wa.asr = &(*corpus_iter);
104  wa.tf = rule_vsr_iter->second.no_of_occurrences;
105  index_iter->second.push_back(wa);
106  }
107  }
108  }
109 
110  //compute IDF
111  int rule_corpus_size = rule_corpus.size();
112  for(inv_index_type::iterator ind_it = inverted_index.begin(); ind_it != inverted_index.end(); ind_it++)
113  {
114  token_idf[*(ind_it->first.token)] = rule_corpus_size / (ind_it->second.size());
115 
116  }
117 
118  //document length
119  compute_document_length();
120 }

134 void asferindexer::compute_document_length()
135 {
136  for(inv_index_type::iterator index_it2 = inverted_index.begin(); index_it2 != inverted_index.end(); index_it2++)
137  {
138  int idf = index_it2->first.idf;
139  for(std::list<weighted_asferrule>::iterator rule_it2 = index_it2->second.begin(); rule_it2 != index_it2->second.end(); rule_it2++)
140  {
141  rule_it2->asr->rule_length += idf * idf * rule_it2->tf * rule_it2->tf;
142  }
143  }
144  for(std::list<asferrule>::iterator corpus_it = rule_corpus.begin(); corpus_it != rule_corpus.end(); corpus_it++)
145  {
146  corpus_it->rule_length = sqrtf(corpus_it->rule_length);
147  }
