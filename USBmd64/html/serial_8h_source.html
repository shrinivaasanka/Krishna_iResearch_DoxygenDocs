388 /*
389  * module_usb_serial_driver() - Helper macro for registering a USB Serial driver
390  * @__serial_drivers: list of usb_serial drivers to register
391  * @__ids: all device ids that @__serial_drivers bind to
392  *
393  * Helper macro for USB serial drivers which do not do anything special
394  * in module init/exit. This eliminates a lot of boilerplate. Each
395  * module may only use this macro once, and calling it replaces
396  * module_init() and module_exit()
397  *
398  */
399 #define usb_serial_module_driver(__name, __serial_drivers, __ids) \
400 static int __init usb_serial_module_init(void) \
401 { \
402  return usb_serial_register_drivers(__serial_drivers, \
403  __name, __ids); \
404 } \
405 module_init(usb_serial_module_init); \
406 static void __exit usb_serial_module_exit(void) \
407 { \
408  usb_serial_deregister_drivers(__serial_drivers); \
409 } \
410 module_exit(usb_serial_module_exit);
