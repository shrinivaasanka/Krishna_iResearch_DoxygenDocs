221 /*
222  * USB Resume Timer: Every Host controller driver should drive the resume
223  * signalling on the bus for the amount of time defined by this macro.
224  *
225  * That way we will have a 'stable' behavior among all HCDs supported by Linux.
226  *
227  * Note that the USB Specification states we should drive resume for *at least*
228  * 20 ms, but it doesn't give an upper bound. This creates two possible
229  * situations which we want to avoid:
230  *
231  * (a) sometimes an msleep(20) might expire slightly before 20 ms, which causes
232  * us to fail USB Electrical Tests, thus failing Certification
233  *
234  * (b) Some (many) devices actually need more than 20 ms of resume signalling,
235  * and while we can argue that's against the USB Specification, we don't have
236  * control over which devices a certification laboratory will be using for
237  * certification. If CertLab uses a device which was tested against Windows and
238  * that happens to have relaxed resume signalling rules, we might fall into
239  * situations where we fail interoperability and electrical tests.
240  *
241  * In order to avoid both conditions, we're using a 40 ms resume timeout, which
242  * should cope with both LPJ calibration errors and devices not following every
243  * detail of the USB Specification.
244  */
245 #define USB_RESUME_TIMEOUT 40 /* ms */

444 /*
445  * USB 3.0 Link Power Management (LPM) parameters.
446  *
447  * PEL and SEL are USB 3.0 Link PM latencies for device-initiated LPM exit.
448  * MEL is the USB 3.0 Link PM latency for host-initiated LPM exit.
449  * All three are stored in nanoseconds.
450  */
451 struct usb3_lpm_parameters {
452  /*
453  * Maximum exit latency (MEL) for the host to send a packet to the
454  * device (either a Ping for isoc endpoints, or a data packet for
455  * interrupt endpoints), the hubs to decode the packet, and for all hubs
456  * in the path to transition the links to U0.
457  */
458  unsigned int mel;
459  /*
460  * Maximum exit latency for a device-initiated LPM transition to bring
461  * all links into U0. Abbreviated as "PEL" in section 9.4.12 of the USB
462  * 3.0 spec, with no explanation of what "P" stands for. "Path"?
463  */
464  unsigned int pel;
465 
466  /*
467  * The System Exit Latency (SEL) includes PEL, and three other
468  * latencies. After a device initiates a U0 transition, it will take
469  * some time from when the device sends the ERDY to when it will finally
470  * receive the data packet. Basically, SEL should be the worse-case
471  * latency from when a device starts initiating a U0 transition to when
472  * it will get data.
473  */
474  unsigned int sel;
475  /*
476  * The idle timeout value that is currently programmed into the parent
477  * hub for this device. When the timer counts to zero, the parent hub
478  * will initiate an LPM transition to either U1 or U2.
479  */
480  int timeout;
481 };

1787 /*
1788  * For various legacy reasons, Linux has a small cookie that's paired with
1789  * a struct usb_device to identify an endpoint queue. Queue characteristics
1790  * are defined by the endpoint's descriptor. This cookie is called a "pipe",
1791  * an unsigned int encoded as:
1792  *
1793  * - direction: bit 7 (0 = Host-to-Device [Out],
1794  * 1 = Device-to-Host [In] ...
1795  * like endpoint bEndpointAddress)
1796  * - device address: bits 8-14 ... bit positions known to uhci-hcd
1797  * - endpoint: bits 15-18 ... bit positions known to uhci-hcd
1798  * - pipe type: bits 30-31 (00 = isochronous, 01 = interrupt,
1799  * 10 = control, 11 = bulk)
1800  *
1801  * Given the device address and endpoint descriptor, pipes are redundant.
1802  */
1803 
1804 /* NOTE: these are not the standard USB_ENDPOINT_XFER_* values!! */
1805 /* (yet ... they're the values used by usbfs) */
1806 #define PIPE_ISOCHRONOUS 0
1807 #define PIPE_INTERRUPT 1
1808 #define PIPE_CONTROL 2
1809 #define PIPE_BULK 3
1810 
1811 #define usb_pipein(pipe) ((pipe) & USB_DIR_IN)
1812 #define usb_pipeout(pipe) (!usb_pipein(pipe))
1813 
1814 #define usb_pipedevice(pipe) (((pipe) >> 8) & 0x7f)
1815 #define usb_pipeendpoint(pipe) (((pipe) >> 15) & 0xf)
1816 
1817 #define usb_pipetype(pipe) (((pipe) >> 30) & 3)
1818 #define usb_pipeisoc(pipe) (usb_pipetype((pipe)) == PIPE_ISOCHRONOUS)
1819 #define usb_pipeint(pipe) (usb_pipetype((pipe)) == PIPE_INTERRUPT)
1820 #define usb_pipecontrol(pipe) (usb_pipetype((pipe)) == PIPE_CONTROL)
1821 #define usb_pipebulk(pipe) (usb_pipetype((pipe)) == PIPE_BULK)

1824  unsigned int endpoint)
1825 {
